Для того, чтобы сделать преобразование XML в JSON можно воспользоваться следующими библиотеками XStream, Jackson. Также можно воспользоваться библиотеками xml2json и json2xml. С помощью них преобразование можно выполнить следующим образом: 

					var xml = 'text',
					    dom = parseXml(xml),
					    json = xml2json(dom),
					    xml2 = json2xml(eval(json));

Еще можно использовать библиотеку https://github.com/abdmob/x2js

					var x2js = new X2JS();
					function convertXml2JSon() {
					    $("#jsonText").val(JSON.stringify(x2js.xml_str2json($("#xmlText").val())));
					}

					function convertJSon2XML() {
					    $("#xmlText").val(x2js.json2xml_str($.parseJSON($("#jsonText").val())));
					}

					convertXml2JSon();
					convertJSon2XML();
					$("#convertToJsonBtn").click(convertXml2JSon);
					$("#convertToXmlBtn").click(convertJSon2XML);
